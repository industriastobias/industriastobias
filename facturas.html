function descargarPDF() { const invoice = document.getElementById("invoice"); const buttons = invoice.querySelectorAll("button"); // Ocultar los botones antes de capturar la imagen buttons.forEach(button => button.style.display = "none"); html2canvas(invoice).then((canvas) => { const imgData = canvas.toDataURL("image/png"); const { jsPDF } = window.jspdf; const doc = new jsPDF("p", "mm", "a4"); const imgWidth = 190; const imgHeight = (canvas.height * imgWidth) / canvas.width; doc.addImage(imgData, "PNG", 10, 10, imgWidth, imgHeight); doc.save("factura_tobias_jewelry.pdf"); // Mostrar los botones nuevamente despuÃ©s de la descarga buttons.forEach(button => button.style.display = "inline-block"); }); }
